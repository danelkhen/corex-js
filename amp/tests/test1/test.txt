(ctl,data) => div()
    contacts => repeater(data.contacts)
        contact => div()
            label()
                document.createTextNode(contact.name)
            input ( { val: contact.name } )
            repeater(contact.phones)
                phone => div()
                    input ( { val: phone.number } )
                    a( { attr: { href: "http://"+phone.number+".com" , text:"hello"} } )
